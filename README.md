## Usage

```tsx
import * as React from 'react';

import MyComponent from 'react-scaling-scroll';

class Example extends React.Component {
    render () {
        return (
            <MyComponent elements={[
                <p key={1}>1</p>,
                <p key={2}>2</p>,
                <p key={3}>3</p>
            ]} />
        )
    }
}
```
